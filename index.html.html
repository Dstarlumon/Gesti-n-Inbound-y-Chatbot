1. Conectar con Google Sheets

Pega aquí la URL de tu Web App de Google Apps Script para conectar la aplicación con tu hoja de cálculo.

2. Instrucciones de Configuración

Paso 1: Crea una nueva hoja de Google Sheets con estos encabezados en la fila 1:

ID | FECHA | NOMBRE | TELEFONO | DOCUMENTO | CORREO | VALOR POLIZA | MEDIO | TIPIFICACION | BASE | AGENTE | OBSERVACION

Paso 2: En Google Sheets, ve a Extensiones → Apps Script y pega el siguiente código:

// ===== CÓDIGO PARA GOOGLE APPS SCRIPT =====
// Pega esto en Extensiones > Apps Script de tu Google Sheet

const SHEET_NAME = "Hoja1"; // Cambia si tu hoja tiene otro nombre

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(10000);

  try {
    const action = e.parameter.action;

    if (action === "ping") {
      return jsonResponse({ status: "ok", message: "Conexión exitosa" });
    }

    if (action === "add") {
      return addRow(e);
    }

    if (action === "getAll") {
      return getAllRows(e);
    }

    if (action === "delete") {
      return deleteRow(e);
    }

    return jsonResponse({ status: "error", message: "Acción no reconocida" });
  } catch (err) {
    return jsonResponse({ status: "error", message: err.toString() });
  } finally {
    lock.releaseLock();
  }
}

function addRow(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  const newId = lastRow > 1 ? Number(sheet.getRange(lastRow, 1).getValue()) + 1 : 1;

  const data = JSON.parse(e.parameter.data);
  sheet.appendRow([
    newId,
    data.fecha,
    data.nombre,
    data.telefono,
    data.documento,
    data.correo,
    data.valor,
    data.medio,
    data.tipificacion,
    data.base,
    data.agente,
    data.observacion
  ]);

  return jsonResponse({ status: "ok", id: newId });
}

function getAllRows(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const rows = [];

  for (let i = 1; i < data.length; i++) {
    const row = {};
    headers.forEach((h, j) => { row[h] = data[i][j]; });
    rows.push(row);
  }

  return jsonResponse({ status: "ok", data: rows });
}

function deleteRow(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const id = Number(e.parameter.id);
  const data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (Number(data[i][0]) === id) {
      sheet.deleteRow(i + 1);
      return jsonResponse({ status: "ok" });
    }
  }

  return jsonResponse({ status: "error", message: "ID no encontrado" });
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

3. Actualización diaria automática

Dado que Google Sheets es la base de datos, los datos se actualizan en tiempo real cada vez que un agente guarda un registro. No necesitas hacer nada adicional. Todos los agentes que abran esta aplicación y la conecten a la misma hoja verán los mismos datos.

Para respaldar los datos automáticamente cada día, puedes agregar este trigger en Apps Script (menú Activadores):

// Agrega esto al mismo Apps Script para respaldo diario
function backupDaily() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  const today = Utilities.formatDate(new Date(), "America/Bogota", "yyyy-MM-dd");
  const backup = sheet.copyTo(ss);
  backup.setName("Respaldo_" + today);
}

Luego en Apps Script ve a Activadores (reloj) → Agrega uno nuevo → Función: backupDaily → Temporizador diario → Hora que prefieras.
